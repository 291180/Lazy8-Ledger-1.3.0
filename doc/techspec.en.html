Technical specification

1.  Written in JAVA version 1.4
2.  Database is McKoi.
3.  All fields have been tested for 100% unicode compatibility.  This program will run any language in the world.
4.  The following tables with there SQL definitions are used:


create table Activity
(CompId INTEGER NOT NULL,
Act_id INTEGER NOT NULL,
RegDate DATE,
InvDate DATE,
Notes VARCHAR(200),
FileInfo VARCHAR(200),
PRIMARY KEY(Act_id,CompId))",

create table Amount
(CompId INTEGER NOT NULL,
Act_id INTEGER NOT NULL,
Account INTEGER NOT NULL,
Amount DOUBLE,
IsDebit INTEGER,
Customer INTEGER,
Notes VARCHAR(40)) " ,

create table Account
(CompId int NOT NULL,
Account int NOT NULL,
AccDesc VARCHAR(255),
IsAsset int,
PRIMARY KEY(Account,CompId))",

create table Customer
(CompId int NOT NULL,
CustId int NOT NULL,
CustName VARCHAR(100),
CustDesc VARCHAR(200),
PRIMARY KEY (CustId,CompId))",

create table UniqNum
(CompId int NOT NULL,
UniqName varchar(50) NOT NULL,
LastNumber int,
PRIMARY KEY (UniqName,CompId))",

create table AccountingPeriods
(CompId int NOT NULL,
StartPeriod date NOT NULL,
EndPeriod date NOT NULL,
PRIMARY KEY (CompId,StartPeriod,EndPeriod))",

create table Company
(CompId int NOT NULL,
Name VARCHAR(200),
PRIMARY KEY (CompId))"};
